def get_latest(df):
    latest = []
    n = len(df)
    for i in range(n):
        row = df.iloc[i,:]
        m = len(row)-1
        j= 0
        while math.isnan(row[m-1-j])&(j<m-5):
            j= j+1
        latest.append(row[m-1-j])
    return latest
